数据库编程

发表于 2021-03-24 更新于 2021-04-16 分类于 数据库学习

1、语句块

declare
- 定义，包括变量、游标等
i int;
i int:=5; //初始化
i:=5; //定义内使用赋值报错
- 动态类型
mj student.smajor%type; //使用表属性类型定义变量
line student%rowtype //使用行类型

begin
- 语句
dems_output.put_line(i);

exception
- 异常处理（预定义异常）
No_data_found/Too_many_rows
函数表达式 into 变量 不会报错
属性值 into 变量 显示异常

end;

2、赋值语句into

- i := 10;
- 查询赋值
    select sage,sclass into i,j from student where sname='张三';
    select * into line from student where sname='张三';
- into拒绝空赋值

3、输出

- dbms_output.put_line();

4、输入(&)

declare
    age student.sage%type;
    name student.sname%type:='&请输入学生姓名';        //
begin
    select sage into age from student where sname='&张三';        //
    dbms_output.put_line(age);
end;

6、循环语句

while循环
    while 条件 loop 
        循环体
    end loop;

for each循环
    declare
        line student%rowtype;
    begin
        for line in (select * from studnet) loop
            dbms_output.put_line(line.sname)
        end loop;
    end;

7、自定义异常

declare
    exception_name exception
begin
    if sql%notfound then            // notfound&found在任何一次操作后重新赋值
        raise exception_name
    end if;
exception
    when exception_name then
        ...
end;

    - 只要查询就更新    
        sql%rowcount
